class Solution:
    def depthSum(self, nestedList: List[NestedInteger]) -> int:
        
        def dfs(nestedList , depth):
        
        
        
        
            ans = 0 
            for val in nestedList : 
        
                if val.isInteger() : 
                    ans += depth *val.getInteger()

                else : 

                    ans += dfs(val.getList() , depth + 1)
                    
            return ans 
                    
        return dfs(nestedList , 1)
